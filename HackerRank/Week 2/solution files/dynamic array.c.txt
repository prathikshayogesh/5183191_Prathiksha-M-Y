#include <stdio.h>
#include <stdlib.h>

typedef struct {
    int *data;
    int size;
    int capacity;
} DynamicArray;

void append(DynamicArray *arr, int value) {
    if (arr->size == arr->capacity) {
        arr->capacity = arr->capacity == 0 ? 1 : arr->capacity * 2;
        arr->data = realloc(arr->data, arr->capacity * sizeof(int));
    }
    arr->data[arr->size++] = value;
}

int main() {
    int n, q;
    scanf("%d %d", &n, &q);

   
    DynamicArray *seqList = calloc(n, sizeof(DynamicArray));
    int lastAnswer = 0;

    for (int i = 0; i < q; i++) {
        int type, x, y;
        scanf("%d %d %d", &type, &x, &y);
        int idx = (x ^ lastAnswer) % n;

        if (type == 1) {
            append(&seqList[idx], y);
        } else if (type == 2) {
            int val = seqList[idx].data[y % seqList[idx].size];
            lastAnswer = val;
            printf("%d\n", lastAnswer);
        }
    }

    
    for (int i = 0; i < n; i++) {
        free(seqList[i].data);
    }
    free(seqList);

    return 0;
}
