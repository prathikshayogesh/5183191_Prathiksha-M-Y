#include <stdio.h>

void miniMaxSum(int arr[], int size) {
    long long totalSum = 0;
    int min = arr[0];
    int max = arr[0];

    for(int i = 0; i < size; i++) {
        totalSum += arr[i];

        if(arr[i] < min) {
            min = arr[i];
        }

        if(arr[i] > max) {
            max = arr[i];
        }
    }

    long long minSum = totalSum - max;
    long long maxSum = totalSum - min;

    printf("%lld %lld\n", minSum, maxSum);
}

int main() {
    int arr[5];

  
    for(int i = 0; i < 5; i++) {
        scanf("%d", &arr[i]);
    }

    miniMaxSum(arr, 5);

    return 0;
}
